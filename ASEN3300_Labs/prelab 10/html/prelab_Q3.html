Create Constants

R = 33e3;
C1 = 0.01e-6;
C2 = 100e-12;
Q = 0.5*sqrt(C1/C2);
omega_0 = 1/(R*sqrt(C1*C2));
freq = 0:1:60e3;
omega = 2*pi*freq;

Get the Transfer Function Magnitude and Phase

transfer = (omega_0^2)./((omega_0^2 -omega.^2) + (1i*omega_0*omega/Q));
mag = abs(transfer);
phase = (180/pi)*angle(transfer);

Create all plots

figure
plot(freq,10*log10(mag))
ylabel('Magnitude [dB]')
xlabel('Frequency [Hz]')
title('Filter Magnitude Response');
grid
axis tight

figure
plot(freq,phase)
ylabel('Phase [deg]')
xlabel('Frequency [Hz]')
title('Filter Phase Response')
grid
axis tight
